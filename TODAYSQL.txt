select dayname('1995/07/01') as day_of_dob from dual;

select ename , hiredate , dayname(hiredate) as day_name from emp where dayname(hiredate) = 'thursday';

SELECT e.empno, e.ename, e.deptno, e.sal, s.grade, d.dname
FROM emp e
JOIN dept d ON e.deptno = d.deptno
JOIN salgrade s ON e.sal BETWEEN s.losal AND s.hisal
WHERE e.sal > (
    SELECT MAX(e2.sal)
    FROM emp e2
    JOIN dept d2 ON e2.deptno = d2.deptno
    WHERE d2.dname = 'sales'
);                                                                                                           

SELECT 
    CASE 
        WHEN (year % 4 = 0 AND year % 100 != 0) OR (year % 400 = 0) 
        THEN 'Leap Year' 
        ELSE 'Not a Leap Year' 
    END AS result
FROM (SELECT 2026 AS year) AS test;
